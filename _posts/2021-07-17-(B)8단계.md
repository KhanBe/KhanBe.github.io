### 1. 손익분기점

```c
#include <stdio.h>

int main() {
	int a, b, c;
	scanf("%d %d %d", &a, &b, &c);

	if (b >= c) {
		printf("-1");
		return 0;
	}

	int gap = c - b;

	int count = a / gap;
	count++;

	printf("%d", count);
}
```

### 2. 벌집

```c
#include <stdio.h>

int main() {
	int n;
	scanf("%d", &n);

	int sum = 6;
	int extent = 1;

	int count = 1;
	while (extent < n) {
		extent += sum;
		sum += 6;

		count++;
	}

	printf("%d", count);
}
```

### 3. 분수찾기

```c
#include <stdio.h>
#include <stdbool.h>

int main() {
	int n;
	scanf("%d", &n);

	bool right = false;

	int sum = 0;
	int count = 0;
	while (n > sum) {
		count++;
		sum += count;

		right = right ? false : true;
	}

	int diff = sum - n;

	int a = 1 + diff;
	int b = count - diff;

	if (right) printf("%d/%d", a, b);
	else printf("%d/%d", b, a);
}
```
