### 1. 손익분기점

```c
#include <stdio.h>

int main(){
    int t;
    scanf("%d",&t);
    
    int count = 0;
    for(int i= 0; i<t;i++){
        int n;
        scanf("%d",&n);
        
        int isPrime = 1;
        
        if(n==1 || n==0) continue;
        else if(n==2){
            count ++;
            continue;
        }
        for(int j=2;j<=n/2;j++){
            if(n%j==0) {
                isPrime = 0;
                break;
            }
        }
        if(isPrime) count ++;
    }
    printf("%d",count);
}
```
