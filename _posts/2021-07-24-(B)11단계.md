### 1. 블랙

```c
#include <stdio.h>

int main(){
    int n,m;
    scanf("%d %d",&n, &m);
    
    int card[n+1];
    for(int i=0;i<n;i++)
        scanf("%d", &card[i]);
    
    int sum = 0;
    int max = 0;
    for(int i=0;i<n-2;i++){
        for(int j=i+1;j<n-1;j++){
            for(int k=j+1;k<n;k++){
                if(i==j || j==k) continue;
                sum = card[i] + card[j] + card[k];
                if(sum > m) continue;
                if(max < sum) max = sum;
            }
        }
    }
    if(max > m) printf("%d",m);
    else printf("%d",max);
}
```
