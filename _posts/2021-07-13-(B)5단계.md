### 1. 최소, 최대

```c
#include <stdio.h>

int main(){
    int n;
    scanf("%d",&n);
    
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    int max;
    int min;
    max = min = arr[0];
    
    for(int i=0;i<n;i++){
        if(arr[i] < min) min = arr[i];
        if(arr[i] > max) max = arr[i];
    }
    printf("%d %d",min,max);
}
```

### 3. 숫자의 개수

```c
#include <stdio.h>

int main(){
    int a,b,c;
    int arr[10] = {};
    
    scanf("%d",&a);
    scanf("%d",&b);
    scanf("%d",&c);
    
    int mul = a*b*c;
    while(mul != 0){
        arr[mul%10]++;
        mul /= 10;
    }
    for(int i=0;i<10;i++){
        printf("%d\n",arr[i]);
    }
}
```

### 5. 평균

```c
#include <stdio.h>

int main(){
    int n;
    scanf("%d",&n);
    
    double arr[n];
    double max = 0;
    for(int i=0;i<n;i++){
        scanf("%lf",&arr[i]);
        if(max < arr[i]) max = arr[i];
    }
    
    double sum = 0;
    for(int i=0;i<n;i++){
        arr[i] = (arr[i]/max)*100;
        sum += arr[i];
    }
    printf("%.2lf", sum/(double)n);
}
```

### 7. 평균은 넘겠지

```c
#include <stdio.h>

int main(){
    int testCase,n;
    scanf("%d",&testCase);
    
    for(int i=0;i<testCase;i++){
        scanf("%d",&n);
        
        double score[n];
        double sum=0;
        for(int j=0;j<n;j++){
            scanf("%lf",&score[j]);
            sum += score[j];
        }
        double avg = sum/(double)n;
        
        double count = 0;
        for(int j=0;j<n;j++){
            if(score[j] > avg) count++;
        }
        printf("%.3f%%\n",count/(double)n*100);
    }
}
```
